//program to demonistrate single inheritance
public class A {
	int p=10;
	void display() {
		System.out.println("value of p:"+p);
	}
	
	
		
	}
	
public class B extends A{
	int q=20;
	
		
	void show() {
		System.out.println("value of q:"+q);
	}
		void add() {
			System.out.println("addition of p and q:"+(p+q));
	}
	}

public class InhertEx {

	public static void main(String[] args) {
		B obj=new B();
		obj.display();
		obj.show();
		obj.add();
			
	}

        }

output:
value of p:10
value of q:20
addition of p and q:30
